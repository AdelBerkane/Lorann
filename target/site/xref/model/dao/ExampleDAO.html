25  	private static int symboleColumnIndex = 2;
26  
27  	/**
28  	 * Gets the level by
29  	 *
30  	 * @param id
31  	 *            the id
32  	 * @return the example by id
33  	 * @throws SQLException
34  	 *             the SQL exception
35  	 */
36  	
37  	// just change <Example> by table name
38  	public static ArrayList<Example> getLevel(final int id1, final int id2) throws SQLException {
39  		final CallableStatement callStatement = prepareCall(sqlLevelbyId);
40  		final ArrayList<Example> examples = new ArrayList<Example>();
41  		callStatement.setInt(1, id1);
42  		callStatement.setInt(2, id2);
43  
44  		if (callStatement.execute()) {
45  			final ResultSet result = callStatement.getResultSet();
46  			for (boolean isResultLeft = result.first(); isResultLeft; isResultLeft = result.next()) {
47  				examples.add(new Example(result.getInt(idColumnIndex), result.getString(symboleColumnIndex)));
48  			}
49  
50  			result.close();
51  		}
52  		return examples;
53  	}
